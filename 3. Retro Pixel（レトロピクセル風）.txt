🧠 Kotlinコード（MainActivity.kt）
package com.example.retropixelui

import android.os.Bundle
import android.widget.Button
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity

class MainActivity : AppCompatActivity() {

    private lateinit var file1: Button
    private lateinit var file2: Button
    private lateinit var file3: Button
    private lateinit var selectButton: Button
    private var selectedFile: String = "ドキュメント2"

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        file1 = findViewById(R.id.file1)
        file2 = findViewById(R.id.file2)
        file3 = findViewById(R.id.file3)
        selectButton = findViewById(R.id.selectButton)

        file1.setOnClickListener {
            selectedFile = "ドキュメント1"
            highlightSelected(file1)
        }

        file2.setOnClickListener {
            selectedFile = "ドキュメント2"
            highlightSelected(file2)
        }

        file3.setOnClickListener {
            selectedFile = "ドキュメント3"
            highlightSelected(file3)
        }

        selectButton.setOnClickListener {
            Toast.makeText(this, "$selectedFile を選択しました", Toast.LENGTH_SHORT).show()
        }
    }

    private fun highlightSelected(selectedButton: Button) {
        val buttons = listOf(file1, file2, file3)
        buttons.forEach {
            it.setBackgroundColor(android.graphics.Color.parseColor("#001100"))
            it.setTextColor(android.graphics.Color.parseColor("#00FF00"))
        }
        selectedButton.setBackgroundColor(android.graphics.Color.parseColor("#00FF00"))
        selectedButton.setTextColor(android.graphics.Color.parseColor("#002200"))
    }
}
